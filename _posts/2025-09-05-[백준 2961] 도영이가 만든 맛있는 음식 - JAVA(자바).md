### 해결방법
- **백트래킹** 활용하기
    - 꽃을 심을 수 있는 위치는 (1,1)부터 (n-2,n-2)까지
    - 꽃을 심을 때마다, 해당 위치와 상하좌우 위치를 방문 처리
    - 꽃을 3개 심었을 때, 최소 비용 갱신
    - 모든 위치를 탐색하면서, 최소 비용 갱신


<br /><br />

### 코드

```java
import java.util.*;
import java.lang.*;
import java.io.*;

class Main {
    static int n;
    static int[][] flowerBed;
    static boolean[][] visited;
    static int[] dx = {1, -1, 0, 0};
    static int[] dy = {0, 0, 1, -1};
    static int min = Integer.MAX_VALUE;
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        n = Integer.parseInt(br.readLine());
        flowerBed = new int[n][n];
        
        for (int i = 0; i < n; i++) {
            StringTokenizer st = new StringTokenizer(br.readLine());
            for (int j = 0; j < n; j++) {
                flowerBed[i][j] = Integer.parseInt(st.nextToken());
            }
        }

        visited = new boolean[n][n];
        dfs(0, 0);
        
        System.out.println(min);
    }

    private static void dfs(int depth, int cost) {
        if (depth == 3) { // 씨앗 3개를 다 심었으면 최소값 갱신 후 종료
            min = Math.min(min, cost);
            return;
        }

        for (int i = 1; i < n-1; i++) {
            for (int j = 1; j < n-1; j++) {
                // 현재 좌표를 중심으로 5개의 좌표가 모두 비었는지 확인
                if (visited[i][j]) continue;
                boolean flag = true;
                for (int k = 0; k < 4; k++) {
                    int nx = i + dx[k];
                    int ny = j + dy[k];
                    if (visited[nx][ny]) {
                        flag = false;
                        break;
                    }
                }
                if (!flag) continue;

                // 5개의 좌표 모두 방문처리 및 비용 계산
                int sum = flowerBed[i][j];
                visited[i][j] = true;
                for (int k = 0; k < 4; k++) {
                    int nx = i + dx[k];
                    int ny = j + dy[k];
                    sum += flowerBed[nx][ny];
                    visited[nx][ny] = true;
                }
                dfs(depth+1, cost+sum);
                // 방문처리 해제
                visited[i][j] = false;
                for (int k = 0; k < 4; k++) {
                    int nx = i + dx[k];
                    int ny = j + dy[k];
                    visited[nx][ny] = false;
                }      
            }
        }

        
    }
}
```
